formulario ::= formulario:n1 tabla:n2								{: RESULT = n1; n1.add(n2); :}
	| formulario:n1 seccion:n2								{: RESULT = n1; n1.add(n2); :}
	| tabla:n1										{: RESULT = new LinkedList<Nodoui>(); RESULT.add(n1); :}
	| seccion:n1										{: RESULT = new LinkedList<Nodoui>(); RESULT.add(n1); :}
	;
	
pregunta ::= pregunta_abierta:n1								{: RESULT = n1; :}
	| pregunta_multiple:n1									{: RESULT = n1; :}
	| pregunta_select:n1									{: RESULT = n1; :}
	| pregunta_drop:n1									{: RESULT = n1; :}
	;
	
lista_correct ::= LLAVEAPER numeros:n2 LLAVECERRA						{: RESULT = n2; :}
	;
	
numeros ::= numeros:n1 COMA NUMERO:n3								{: RESULT = n1; n1.add(n3); :}
	| NUMERO:n1										{: RESULT = new LinkedList<Double>(); RESULT.add(n1); :}
	|											{: RESULT = new LinkedList<Double>(); :}
	;
	
pregunta_multiple ::= pregmul_aper:n1 pregmul_final:n2						{: RESULT = n1; n1.setEstilo(n2); :}
	;
	
pregmul_aper ::= APERT MULTIPLE IGUAL NUMERO:n4 COMA 
	NUMERO:n6 COMA CADENA:n8 COMA lista_op:n10 COMA lista_correct:n12			{: RESULT = new PreguntaMultiple(n4, n6, n8, n10, n12); :}
	;
	
pregmul_final ::= DIAGON CERRA
	| CERRA estilo:n2 APERT DIAGON MULTIPLE CERRA						{: RESULT = n2; :}
	;
	
pregunta_select ::= pregsel_aper:n1 pregsel_final:n2						{: RESULT = n1; n1.setEstilo(n2); :}
	;
	
pregsel_aper ::= APERT SELECT IGUAL NUMERO:n4 COMA NUMERO:n6 COMA CADENA:n8 COMA 
	lista_op:n10 COMA NUMERO:n12								{: RESULT = new PreguntaSeleccion(n4, n6, n8, n10, n12); :}
	;
	
pregsel_final ::= DIAGON CERRA
	| CERRA estilo:n2 APERT DIAGON SELECT CERRA						{: RESULT = n2; :}
	;
	
lista_op ::= LLAVEAPER opciones:n2 LLAVECERRA							{: RESULT = n2; :}
	;
	
opciones ::= opciones:n1 COMA CADENA:n3								{: RESULT = n1; n1.add(n3); :}
	| CADENA:n1										{: RESULT = new LinkedList<String>(); RESULT.add(n1); :}
	;
	
pregunta_drop ::= predrop_aper:n1 predrop_final:n2						{: RESULT = n1; n1.setEstilo(n2); :}
	;
	
predrop_aper ::= APERT DROP IGUAL NUMERO:n4 COMA NUMERO:n6 COMA CADENA:n7 COMA 
	lista_op:n9 COMA NUMERO:n11								{: RESULT = new PreguntaDesplegable(n4, n6, n7, n9, n11); :}
	;

	| CERRA estilo:n2 APERT DIAGON DROP CERRA						{: RESULT = n2; :}
	;
